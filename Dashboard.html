function parsearFecha(fechaStr){ if(!fechaStr) return null; // Formato DD/MM/YYYY const dmy = fechaStr.match(/^(\d{2})\/(\d{2})\/(\d{4})$/); if(dmy) return new Date(`${dmy[3]}-${dmy[2]}-${dmy[1]}`); // Formato YYYY-MM-DD const ymd = fechaStr.match(/^(\d{4})-(\d{2})-(\d{2})$/); if(ymd) return new Date(`${ymd[1]}-${ymd[2]}-${ymd[3]}`); return new Date(fechaStr); } function renderizar(datos){ let vig=0,pv=0,ven=0; let holdCount={}; const tbody=document.getElementById("tableBody"); tbody.innerHTML=""; datos.forEach(d=>{ const estado=calcularEstado(d["Fecha_Vencimiento"]); if(estado==="Vigente") vig++; if(estado==="Por Vencer") pv++; if(estado==="Vencida") ven++; if(d.Comentarios && d.Comentarios.trim()!==""){ holdCount[d.Comentarios]=(holdCount[d.Comentarios]||0)+1; } tbody.innerHTML+=` ${d.Nombre||""} ${d["SAP ID"]||""} ${d.Certificacion||""} ${d["Activity Type"]||""} ${d["Fecha_Certificacion"]||""} ${d["Fecha_Vencimiento"]||""} ${estado} ${d.Comentarios||""} `; }); document.getElementById("vigentes").textContent=vig; document.getElementById("porVencer").textContent=pv; document.getElementById("vencidas").textContent=ven; document.getElementById("total").textContent=datos.length; // Charts if(estadoChart) estadoChart.destroy(); estadoChart=new Chart(document.getElementById("estadoChart"),{ type:"doughnut", data:{labels:["Vigente","Por Vencer","Vencida"], datasets:[{data:[vig,pv,ven]}]} }); if(holdChart) holdChart.destroy(); holdChart=new Chart(document.getElementById("holdChart"),{ type:"bar", data:{labels:Object.keys(holdCount), datasets:[{data:Object.values(holdCount)}]} }); }
